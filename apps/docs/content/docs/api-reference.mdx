## `convert(input, options?)`

```ts
function convert(
  input: string | Buffer | Uint8Array | ArrayBuffer,
  options?: ConvertOptions,
): Promise<ConvertedPage[]>;
```

## Inputs

<TypeTable
  type={{
    "file path": {
      type: "string",
      description: "Absolute or relative path to a PDF on disk.",
    },
    Buffer: {
      type: "Buffer",
      description: "PDF bytes already loaded into memory.",
    },
    Uint8Array: {
      type: "Uint8Array",
      description: "Typed-array PDF bytes.",
    },
    ArrayBuffer: {
      type: "ArrayBuffer",
      description: "Browser-style binary buffer passed from server code.",
    },
  }}
/>

## `ConvertOptions`

<TypeTable
  type={{
    pages: {
      type: "number[]",
      description: "Zero-based page indices to render.",
      required: false,
      default: "all pages",
    },
    dpi: {
      type: "number",
      description: "Target raster DPI.",
      required: false,
      default: "300",
    },
    outputFormat: {
      type: '"png"',
      description: "Output format. PNG is the only supported value in v1.",
      required: false,
      default: '"png"',
    },
    password: {
      type: "string",
      description: "Password for encrypted PDFs.",
      required: false,
    },
    crop: {
      type: "{ x; y; width; height }",
      description: "Crop rectangle applied after rendering.",
      required: false,
      typeDescription:
        "{ x: number; y: number; width: number; height: number }",
    },
    renderAnnotations: {
      type: "boolean",
      description: "Whether annotations and form data should be rendered.",
      required: false,
      default: "true",
    },
  }}
/>

<Callout type="warning" title="Page indices are zero-based">
  `pages: [0]` means the first page. `pages: [1]` means the second page.
</Callout>
